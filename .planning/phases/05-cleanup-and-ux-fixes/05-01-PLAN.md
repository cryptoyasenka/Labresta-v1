2. **Add inline global badge polling + dropdown JS** after the `common.js` script tag (line 117) and BEFORE the `{% block scripts %}` tag. Wrap in `{% if current_user.is_authenticated %}` block:

```html
{% if current_user.is_authenticated %}
<script>
(function() {
    // --- Global badge polling (runs on every page, every 30s) ---
    function updateNavbarBadge() {
        var badge = document.getElementById('notificationBadge');
        if (!badge) return;
        fetch('/settings/api/notifications/unread')
            .then(function(r) { return r.json(); })
            .then(function(data) {
                var count = data.length || 0;
                var badgeText = badge.childNodes[0]; // text node before the dismiss button
                if (count > 0) {
                    // Update text content without destroying the dismiss button
                    if (badgeText && badgeText.nodeType === 3) {
                        badgeText.textContent = count + ' ';
                    } else {
                        badge.insertBefore(document.createTextNode(count + ' '), badge.firstChild);
                    }
                    badge.style.display = '';
                } else {
                    badge.style.display = 'none';
                }
            })
            .catch(function() {});
    }
    // Expose globally so notifications.js page-specific code can call it
    window.updateNavbarBadge = updateNavbarBadge;

    updateNavbarBadge();
    setInterval(updateNavbarBadge, 30000);

    // --- Dropdown: load notifications on every open ---
    var dropdownEl = document.getElementById('notificationDropdown');
    if (dropdownEl) {
        dropdownEl.addEventListener('show.bs.dropdown', loadDropdownNotifications);
    }

    function loadDropdownNotifications() {
        var container = document.getElementById('notificationDropdownItems');
        var emptyMsg = document.getElementById('notificationDropdownEmpty');
        if (!container) return;

        fetch('/settings/api/notifications/unread')
            .then(function(r) { return r.json(); })
            .then(function(data) {
                var items = data.slice(0, 5);
                container.innerHTML = '';
                if (items.length === 0) {
                    emptyMsg.style.display = '';
                    return;
                }
                emptyMsg.style.display = 'none';
                items.forEach(function(n) {
                    var li = document.createElement('a');
                    li.href = '#';
                    li.className = 'dropdown-item small notification-item py-2';
                    li.setAttribute('data-id', n.id);
                    li.onclick = function(e) { e.preventDefault(); };
                    li.innerHTML =
                        '<div class="fw-semibold text-truncate" style="max-width:310px;">' +
                        escapeHtml(n.message) + '</div>' +
                        '<small class="text-muted">' + escapeHtml(n.created_at) + '</small>';
                    container.appendChild(li);
                });
            })
            .catch(function() {});
    }

    function escapeHtml(text) {
        var div = document.createElement('div');
        div.textContent = text;
        return div.innerHTML;
    }

    // --- Dropdown: mark all read button ---
    var markBtn = document.getElementById('dropdownMarkAllRead');
    if (markBtn) {
        markBtn.addEventListener('click', function(e) {
            e.preventDefault();
            e.stopPropagation(); // Keep dropdown open
            var items = document.querySelectorAll('#notificationDropdownItems .notification-item');
            var ids = [];
            items.forEach(function(el) {
                var id = el.getAttribute('data-id');
                if (id) ids.push(parseInt(id, 10));
            });
            if (ids.length === 0) return;

            fetchWithCSRF('/settings/api/notifications/mark-read', {
                method: 'POST',
                body: JSON.stringify({ ids: ids }),
            })
            .then(function(r) { return r.json(); })
            .then(function(data) {
                if (data.marked > 0) {
                    updateNavbarBadge();
                    // Refresh dropdown content
                    loadDropdownNotifications();
                }
            })
            .catch(function() {});
        });
    }
})();
</script>
{% endif %}
```

**Key implementation details:**
- The dismiss (x) button is INSIDE the badge span. `event.stopPropagation()` prevents it from toggling the dropdown.
- `show.bs.dropdown` event fires on EVERY dropdown open, ensuring fresh data each time (avoids Pitfall 5).
- `updateNavbarBadge` is assigned to `window.updateNavbarBadge` so notifications.js page-specific mark-all-read handler can still call it.
- Script is placed AFTER `common.js` (which provides `fetchWithCSRF`) and AFTER the navbar HTML (which contains `#notificationBadge`), avoiding Pitfall 2.
- The badge text update preserves the dismiss button child element by only modifying the text node.
